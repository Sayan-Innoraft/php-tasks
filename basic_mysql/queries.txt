1. select employee_first_name from employee_details inner join
employee_salary_table on employee_details.employee_id = employee_salary_table
.employee_id where employee_salary > 50000;

2. select employee_last_name from employee_details WHERE graduation_percentile >
 70;

3. select employee_code_name from employee_code_table INNER JOIN
(employee_salary_table INNER JOIN employee_details on employee_salary_table
.employee_id= employee_details.employee_id) on employee_salary_table
.employee_code = employee_code_table.employee_code where graduation_percentile <
 70;

4. select concat(employee_first_name,' ',employee_last_name) as 'Full Name' from
 employee_details INNER JOIN (employee_salary_table INNER JOIN
 employee_code_table on employee_salary_table.employee_code =
 employee_code_table.employee_code) on employee_details.employee_id =
 employee_salary_table.employee_id where employee_code_table.employee_domain !=
 'Java';

5. select employee_domain,sum(employee_salary) as 'sum_of_salary' from
employee_salary_table inner join employee_code_table on employee_salary_table
.employee_code = employee_code_table.employee_code group by employee_domain;

6. select employee_domain,sum(employee_salary) as 'sum_of_salary' from
employee_salary_table inner join employee_code_table on employee_salary_table
.employee_code = employee_code_table.employee_code where employee_salary > 30000
 group by employee_domain ;

7. select employee_id from employee_salary_table where employee_code is null;
